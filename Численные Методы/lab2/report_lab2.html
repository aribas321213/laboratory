def simple_iteration(f: Callable[[float], float], 
                     phi: Callable[[float], float],
                     x0:float, 
                     eps=1e-8, 
                     max_iter=1000
)-> tuple[float, list[float]]:
    """
      

    Args:
        f (Callable[[float], float]):  
        phi (Callable[[float], float]):     
        x0 (float):   
        eps (_type_, optional):  . Defaults to 1e-8.
        max_iter (int, optional):   . Defaults to 1000.

    Returns:
        tuple[float, list[float]]:     
    """

    x = x0
    resids = []
    for iter in range(max_iter):
        x_new = phi(x)
        resids.append(abs(x - x_new))
        if abs(x_new - x) < eps:
            return x, resids
        x = x_new

    return x, resids

def newton_method(f: Callable[[float], float], 
                  df: Callable[[float], float], 
                  x0: float, 
                  eps=1e-8, 
                  max_iter=100
)-> tuple[float, list[float]]:
    """
     

    Args:
        f (Callable[[float], float]):  
        df (Callable[[float], float]):  
        x0 (float):  
        eps (float, optional):  . Defaults to 1e-8.
        max_iter (int, optional):   . Defaults to 100.

    Returns:
        tuple[float, list[float]]:     
    """

    x = x0
    resids = []
    for i in range(max_iter):
        if df(x) == 0:
            print("  .     .")
            return None, i
        x_new = x - f(x) / df(x)
        resids.append(abs(x - x_new))
        if abs(x_new - x) < eps:
            return x, resids
        x = x_new
    
    return x, resids

def simple_iteration_system(F: Callable[[float,float], list[float]],
                            Phi: Callable[[float,float], list[float]],
                            x0: float,
                            tol=1e-15,
                            max_iter=1000
)-> tuple[list[float], list[float]]:
    """
        

    Args:
        F (Callable[[float,float], list[float]]):  
        Phi (Callable[[float,float], list[float]]):     
        x0 (float):  
        tol (float, optional):  . Defaults to 1e-6.
        max_iter (int, optional):   . Defaults to 1000.

    Returns:
        tuple[list[float], list[float]]:     
    """
    
    x = x0.copy()
    resids = []
    for i in range(max_iter):
        x_new = Phi(x)
        resids.append(np.linalg.norm(x_new - x))
        if np.linalg.norm(x_new - x) < tol:
            return x_new, resids
        x = x_new
    return x, resids

def newton_method_system(F: Callable[[float,float], list[float]],
                         J: Callable[[float,float], list[float]],
                         x0: float,
                         tol=1e-10,
                         max_iter=1000
)-> tuple[list[float], list[float]]:
    """
       

    Args:
        F (Callable[[float,float], list[float]]):  
        J (Callable[[float,float], list[float]]):    
        x0 (float):  
        tol (float, optional):  . Defaults to 1e-10.
        max_iter (int, optional):   . Defaults to 1000.

    Returns:
        tuple[list[float], list[float]]:     
    """

    x = x0.copy()
    resids = []
    for i in range(max_iter):
        J_x = J(x)
        if np.linalg.det(J_x) == 0:
            print("  .     .")
            return None, i
        F_x = F(x)
        delta_x = np.dot(np.linalg.inv(J_x), F_x)
        x_new = x - delta_x
        resids.append(np.linalg.norm(x_new - x))
        if np.linalg.norm(x_new - x) < tol:
            return x_new, resids
        x = x_new
    return x, resids

def F(x):
    return np.array([
            2*(x[0]**2) - x[1] + x[1]**2 - 2,
            x[0] - np.sqrt(x[1]+2) + 1
    ])

def Phi(x):
    return np.array([
                np.sqrt((2 - x[1]**2 + x[1])/2),
                (1 + x[0])**2 - 2
            ])

def J(x):
    return np.array([
        [4*x[0], 2*x[1]-1],
        [1, -1/(2*np.sqrt(x[1]+2))]   
    ])
